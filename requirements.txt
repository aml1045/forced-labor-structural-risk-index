# FLSRI Python dependencies.
#
# Two entrypoints, two dependency footprints:
#
#   run.py                 -> CORE only (stdlib + one optional package)
#   build_site_data.py /   -> FULL (the complete data + geospatial pipeline)
#   pipeline/, validation
#
# Pins below are the versions the 2026-06-09 builds were actually verified
# against (the .venv311 environment, Python 3.11 — see .python-version). The
# byte-identical site-data reproduction documented in docs/REPRODUCING.md was
# produced under exactly these versions.
#
# NON-PYTHON DEPENDENCY: the site-data build (steps 11/15 of
# build_site_data.py) shells out to `npx mapshaper`. mapshaper is pinned in
# package.json (its simplification output is version-dependent); install with
# `npm ci` (Node >= 18).

# ---------------------------------------------------------------------------
# CORE -- what `python run.py` needs.
# run.py runs on the Python standard library alone. pycountry is OPTIONAL:
# it is used only to resolve ISO3 codes to country names, and run.py falls
# back to the bare ISO3 code if it is not installed.
# ---------------------------------------------------------------------------
pycountry==26.2.16         # optional; run.py degrades gracefully without it

# ---------------------------------------------------------------------------
# FULL -- the rest of the pipeline (build_site_data.py, pipeline/*,
# the validation scripts). Install these to ingest real data, build the
# site-data layer, run the spatial analysis, and run validation.
# ---------------------------------------------------------------------------
pandas==3.0.3              # data frames throughout the pipeline
numpy==2.4.6               # numerics
scipy==1.17.1              # stats (validation, KDE/Gi* surfaces)
pyyaml==6.0.3              # config/api-config/*.yaml loaders
openpyxl==3.1.5            # reads .xlsx source workbooks (e.g. GSI)
country_converter==1.2     # country-name -> ISO3 normalization
certifi==2026.5.20         # CA bundle for HTTPS source pulls

# Geospatial (subnational / admin-1 surfaces, LISA / spatial autocorrelation)
geopandas==1.1.3           # vector geo I/O and joins
shapely==2.1.2             # geometry primitives
libpysal==4.14.1           # spatial weights (Queen contiguity, distance)
esda==2.9.0                # exploratory spatial data analysis (Moran's I, LISA)
